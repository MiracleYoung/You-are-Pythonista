def minusOne(num):
    return num - 1

while True:
    try:
        lineNum = int(input())
        temp = []
        for i in range(2, lineNum + 2):
            temp.append(sum(range(i)))
        for j in range(lineNum):
            print(' '.join(map(str, temp)))
            temp = list(map(minusOne, temp[1:]))
    except Exception:
        break
