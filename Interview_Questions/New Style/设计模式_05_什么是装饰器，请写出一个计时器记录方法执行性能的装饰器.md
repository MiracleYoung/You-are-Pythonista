## Analysis
```
import time
from functools import wraps

def timeit(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.clock()
        ret = func(*args, **kwargs)
        end = time.clock()
        print('used:', end - start)
        return ret

    return wrapper

@timeit
def foo():
    print('in foo()')

foo()
```
